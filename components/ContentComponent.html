import { Component, OnInit } from '@angular/core';
import { AuthService } from 'src/app/services/auth.service';
import { CartService } from 'src/app/services/cart.service';
import { ProductService } from '../../services/product.service';

@Component({
  selector: 'app-content',
  templateUrl: './content.component.html',
  styleUrls: ['./content.component.css']
})
export class ContentComponent {
  
  nombre = "";
  userLogged : any;
  constructor(public authService: AuthService,public product:ProductService,private cart:CartService) { 
    // console.log((this.product).ntt);
    this.authService.getUserLogged().subscribe(res=>{
      console.log(res);
      this.userLogged = res;
    });
  }

  traspasoCart(cod:HTMLInputElement,number:HTMLInputElement){

    this.cart.carro(cod.value,number.value);
    
  }

}
